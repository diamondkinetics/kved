89static void kved_print(kved_word_t val)
90{
91 uint8_t *p = (uint8_t *) &val;
92
93 p += KVED_FLASH_WORD_SIZE - 1;
94 for(uint8_t n = 0 ; n < sizeof(kved_word_t) ; n++, p--)
95 printf("%02X",*p);
96}

98static void kved_print_ascii(kved_word_t val, size_t size, bool reverse)
99{
100 uint8_t *p = (uint8_t *) &val;
101
102 if(reverse)
103 p += size - 1;
104
105 for(uint8_t n = 0 ; n < size ; n++)
106 {
107 if(isprint(*p))
108 {
109 printf("%c",*p);
110 }
111 else
112 {
113 printf(".");
114 }
115
116 if(reverse)
117 p--;
118 else
119 p++;
120 }
121}

123static void kved_internal_dump(kved_ctrl_t *ctrl)
124{
125 bool first_free_printed = false;
126 kved_word_t hdr = kved_flash_data_read(ctrl->sector,0);
127 kved_word_t cnt = kved_flash_data_read(ctrl->sector,1);
128
129#if KVED_FLASH_WORD_SIZE == 8
130 printf("HDR (SEC %c) SIGNATURE COUNTER\r\n",(ctrl->sector == 0 ? 'A' : 'B'));
131#else
132 printf("HDR (SEC %c) SIGNAT. COUNTER\r\n",(ctrl->sector == 0 ? 'A' : 'B'));
133#endif
134
135 printf("ITEM IDX TYP SZ ");
136 kved_print(hdr);
137 printf(" ");
138 kved_print(cnt);
139 printf("\r\n");
140
141 for(uint16_t index = ctrl->first_index ; index <= ctrl->last_index && !first_free_printed; index += KVED_ENTRY_SIZE_IN_WORDS)
142 {
143 kved_word_t key = kved_flash_data_read(ctrl->sector,index);
144 kved_word_t val = kved_flash_data_read(ctrl->sector,index + 1);
145
146 if(key == KVED_DELETED_ENTRY)
147 {
148 printf("DEL ");
149 }
150 else if(key == KVED_FREE_ENTRY)
151 {
152 if(val == KVED_FREE_ENTRY)
153 {
154 printf("FREE ");
155 first_free_printed = true;
156 }
157 else
158 {
159 printf("ERR1 ");
160 }
161 }
162 else
163 {
164 printf("USED ");
165 }
166
167 uint8_t size = KVED_HDR_MASK_SIZE(key);
168 uint8_t type = KVED_HDR_MASK_TYPE(key);
169
170 if((val == KVED_FREE_ENTRY) || (key == KVED_DELETED_ENTRY))
171 {
172 printf("%03d ",index);
173 }
174 else
175 {
176 printf("%03d %3s %02d ",index,(char *)kved_data_type_label[type],size);
177 }
178 kved_print(key);
179 printf(" ");
180 kved_print(val);
181 printf(" ");
182 kved_print_ascii(key,KVED_FLASH_WORD_SIZE,true);
183 printf(" ");
184 kved_print_ascii(val,KVED_FLASH_WORD_SIZE,type == KVED_DATA_TYPE_STRING ? false : true);
185 printf("\r\n");
186 }
187
188 printf("TOTAL %d USED %d DELETED %d FREE %d\r\n\r\n",
189 ctrl->stats.num_total_entries,
190 ctrl->stats.num_used_entries,
191 ctrl->stats.num_deleted_entries,
192 ctrl->stats.num_free_entries);
193}

200static void kved_sector_stats_read(kved_ctrl_t *ctrl)
201{
202 // [0,NV_HDR_SIZE] ARE NOT VALID AS ENTRY INDEXES, THEY ARE RESERVED FOR HEADER
203 ctrl->first_index = KVED_HDR_SIZE_IN_WORDS;
204 ctrl->last_index = (kved_flash_sector_size()/KVED_FLASH_WORD_SIZE) - KVED_HDR_SIZE_IN_WORDS;
205 ctrl->first_free_index = 0;
206
207 nv_sector_stats_erase(&ctrl->stats);
208
209 for(uint16_t index = ctrl->first_index ; index <= ctrl->last_index ; index += KVED_ENTRY_SIZE_IN_WORDS)
210 {
211 kved_word_t key = kved_flash_data_read(ctrl->sector,index);
212
213 if(key == KVED_DELETED_ENTRY)
214 {
215 ctrl->stats.num_deleted_entries++;
216 }
217 else if(key == KVED_FREE_ENTRY)
218 {
219 ctrl->stats.num_free_entries++;
220
221 if(ctrl->first_free_index == 0)
222 ctrl->first_free_index = index;
223 }
224 else
225 {
226 ctrl->stats.num_used_entries++;
227 }
228
229 ctrl->stats.num_total_entries++;
230 }
231}

233void kved_key_decode(kved_data_t *data, kved_word_t key)
234{
235 data->type = KVED_HDR_MASK_TYPE(key);
236
237 uint8_t *pkey = (uint8_t *) &key;
238 pkey += KVED_MAX_KEY_SIZE;
239
240 for(size_t p = 0 ; p < KVED_MAX_KEY_SIZE ; p++)
241 data->key[p] = *pkey--;
242}

244kved_word_t kved_key_encode(kved_data_t *data)
245{
246 uint8_t *str_key = data->key;
247 uint8_t key[KVED_MAX_KEY_SIZE];
248
249 strncpy((char *)key,(char *)data->key,KVED_MAX_KEY_SIZE);
250
251 uint8_t size = data->type == KVED_DATA_TYPE_STRING ? strnlen((char *)str_key,KVED_MAX_KEY_SIZE) : kved_data_type_size[data->type];
252 uint8_t hdr = (data->type << 4) | size;
253
254 kved_word_t encoded_key = 0;
255 uint8_t *pkey = (uint8_t *) &encoded_key;
256 pkey += KVED_MAX_KEY_SIZE;
257
258 for(size_t p = 0 ; p < KVED_MAX_KEY_SIZE ; p++)
259 *pkey-- = key[p];
260
261 *pkey = hdr;
262
263 return encoded_key;
264}

266static bool kved_is_valid_key(kved_word_t key)
267{
268 key = KVED_HDR_MASK_KEY(key);
269
270 return (key == KVED_HDR_MASK_KEY(KVED_SIGNATURE_ENTRY)) ||
271 (key == KVED_HDR_MASK_KEY(KVED_DELETED_ENTRY)) ||
272 (key == KVED_HDR_MASK_KEY(KVED_FREE_ENTRY)) ? false : true;
273}

275static uint16_t kved_key_index_find(kved_word_t key)
276{
277 uint16_t key_index = KVED_INDEX_NOT_FOUND;
278
279 key = KVED_HDR_MASK_KEY(key);
280
281 for(uint16_t index = ctrl.first_index ; index <= ctrl.last_index ; index += KVED_ENTRY_SIZE_IN_WORDS)
282 {
283 kved_word_t key_entry = kved_flash_data_read(ctrl.sector,index);
284 key_entry = KVED_HDR_MASK_KEY(key_entry);
285
286 if(key == key_entry)
287 {
288 key_index = index;
289 break;
290 }
291 }
292
293 return key_index;
294}

314static void kved_sector_switch(kved_ctrl_t *ctrl, kved_word_t cnt, kved_word_t upd_key, kved_word_t upd_value)
315{
316 uint16_t next_index = KVED_HDR_SIZE_IN_WORDS;
317 uint16_t total_items = 0;
318 uint16_t used_items = 0;
319 kved_flash_sector_t next_sector = ctrl->sector == KVED_FLASH_SECTOR_A ? KVED_FLASH_SECTOR_B : KVED_FLASH_SECTOR_A;
320
321 kved_flash_sector_erase(next_sector);
322
323 upd_key = KVED_HDR_MASK_KEY(upd_key);
324
325 for(uint16_t index = ctrl->first_index ; index <= ctrl->last_index ; index += KVED_ENTRY_SIZE_IN_WORDS)
326 {
327 kved_word_t key = kved_flash_data_read(ctrl->sector,index);
328
329 if(kved_is_valid_key(key))
330 {
331 kved_word_t val = kved_flash_data_read(ctrl->sector,index + 1);
332
333 kved_flash_data_write(next_sector,next_index++,key);
334
335 if(KVED_HDR_MASK_KEY(key) == upd_key)
336 kved_flash_data_write(next_sector,next_index++,upd_value);
337 else
338 kved_flash_data_write(next_sector,next_index++,val);
339
340 used_items++;
341 }
342
343 total_items++;
344 }
345
346 kved_flash_sector_t last_sector = ctrl->sector;
347 ctrl->sector = next_sector;
348 ctrl->first_index = KVED_HDR_SIZE_IN_WORDS;
349 ctrl->last_index = (kved_flash_sector_size()/KVED_FLASH_WORD_SIZE) - KVED_HDR_SIZE_IN_WORDS;
350 ctrl->first_free_index = next_index;
351 ctrl->stats.num_deleted_entries = 0;
352 ctrl->stats.num_total_entries = total_items;
353 ctrl->stats.num_used_entries = used_items;
354 ctrl->stats.num_free_entries = total_items - used_items;
355
356 // last value is not valid since it is equal to an erased flash entry
357 if((cnt + 1) == KVED_FLASH_UINT_MAX) // last value, avoiding some #if #def related to flash size
358 cnt = 0;
359 else
360 cnt++;
361
362 kved_flash_data_write(next_sector,1,cnt);
363 kved_flash_data_write(next_sector,0,KVED_SIGNATURE_ENTRY);
364
365 kved_flash_data_write(last_sector,0,0); // only invalidate header, it is faster
366}

368static bool kved_internal_data_write(kved_data_t *data)
369{
370 bool sector_changed = false;
371
372 if(!started)
373 return false;
374
375 kved_word_t key = kved_key_encode(data);
376
377 if(!kved_is_valid_key(key))
378 return false;
379
380 uint16_t key_index = kved_key_index_find(key);
381 bool old_entry = key_index != KVED_INDEX_NOT_FOUND;
382
383 // check if the value has changed or not (for existing keys)
384 if(old_entry)
385 {
386 kved_word_t stored_value = kved_flash_data_read(ctrl.sector,key_index + 1);
387
388 if(stored_value == kved_value_encode(data))
389 return true;
390 }
391
392 // no space, exchanging sector do not solve this situation, you need more flash space !
393 if(ctrl.stats.num_total_entries == ctrl.stats.num_used_entries)
394 return false;
395
396 // ok, we have space but a clean up is required before.
397 // Let's do a sector switch and leave the garbage behind.
398 // If we are writing using an existing entries it will
399 // be their valued updated during the process.
400 if(ctrl.stats.num_free_entries == 0)
401 {
402 kved_word_t cnt = kved_flash_data_read(ctrl.sector,1);
403 kved_sector_switch(&ctrl,cnt,key,kved_value_encode(data));
404 sector_changed = true;
405 }
406
407 // AN existing entry is moved to the new sector using the new value already, not to do anymore.
408 // However we need to take care when a new entry is added or when we are updating an entry
409 // in the same sector.
410 bool old_entry_updated_in_the_same_sector = (!sector_changed) && old_entry;
411
412 if(!old_entry || old_entry_updated_in_the_same_sector)
413 {
414 // first data, after key
415 kved_flash_data_write(ctrl.sector,ctrl.first_free_index + 1,kved_value_encode(data));
416 kved_flash_data_write(ctrl.sector,ctrl.first_free_index,key);
417
418 ctrl.stats.num_free_entries--;
419 ctrl.stats.num_used_entries++;
420 ctrl.first_free_index += KVED_ENTRY_SIZE_IN_WORDS;
421
422 // Existing data written in the same sector: erase the old entry
423 if(old_entry_updated_in_the_same_sector)
424 {
425 kved_flash_data_write(ctrl.sector,key_index,KVED_DELETED_ENTRY);
426
427 ctrl.stats.num_deleted_entries++;
428 ctrl.stats.num_used_entries--;
429 }
430 }
431
432#ifdef KVED_DEBUG
433 kved_dump();
434#endif
435
436 return true;
437}

595static bool kved_internal_data_read(kved_data_t *data)
596{
597 if(!started)
598 return false;
599
600 kved_word_t key = kved_key_encode(data);
601
602 if(!kved_is_valid_key(key))
603 return false;
604
605 uint16_t key_index = kved_key_index_find(key);
606
607 if(key_index == KVED_INDEX_NOT_FOUND)
608 return false;
609
610 // update the type as user may not know about them before calling
611 data->type = KVED_HDR_MASK_TYPE(kved_flash_data_read(ctrl.sector,key_index));
612
613 kved_word_t value = kved_flash_data_read(ctrl.sector,key_index+1);
614 kved_value_decode(data,value);
615
616 return true;
617}

668static void kved_internal_format(void)
669{
670 // erase data and control
671 kved_flash_sector_erase(KVED_FLASH_SECTOR_A);
672 kved_flash_sector_erase(KVED_FLASH_SECTOR_B);
673 memset(&ctrl,0,sizeof(ctrl));
674
675 // setup sector as first sector and update stats.
676 // Consistency is not required in such situation.
677 ctrl.sector = KVED_FLASH_SECTOR_A;
678 kved_flash_data_write(ctrl.sector,1,0);// first cnt, after ID
679 kved_flash_data_write(ctrl.sector,0,KVED_SIGNATURE_ENTRY);
680
681 kved_sector_stats_read(&ctrl);
682}

691static void kved_sector_consistency_check(void)
692{
693 bool invalidate_a = false;
694 bool invalidate_b = false;
695
696 kved_word_t id_sec_a = kved_flash_data_read(KVED_FLASH_SECTOR_A,0);
697 kved_word_t id_sec_b = kved_flash_data_read(KVED_FLASH_SECTOR_B,0);
698
699 // Two valid signatures: as the signature is the latest item to be written into the sector
700 // when a formatting or copy operation is performed, probably a restart event happened just
701 // after data copying and, in this case, the section with the
702 // newest cnt will win and the other can be erase as the copy was done.
703 // (remember: last value (0xFF..FF) is not valid as it is the same value of a erased word
704 if((id_sec_a == KVED_SIGNATURE_ENTRY) && (id_sec_b == KVED_SIGNATURE_ENTRY))
705 {
706 kved_word_t cnt_sec_a = kved_flash_data_read(KVED_FLASH_SECTOR_A,1);
707 kved_word_t cnt_sec_b = kved_flash_data_read(KVED_FLASH_SECTOR_B,1);
708
709 // the (a) counter rolled over and the
710 // sector (a->b) copy was done so erase older sector (a) and use the newer (b)
711 if((cnt_sec_a == (KVED_FLASH_UINT_MAX-1)) && (cnt_sec_b == 0))
712 {
713 invalidate_a = true;
714 }
715 // the (b) counter rolled over and the
716 // sector (b->a) copy was done so erase older sector (b) and use the newer (a)
717 else if((cnt_sec_b == (KVED_FLASH_UINT_MAX-1)) && (cnt_sec_a == 0))
718 {
719 invalidate_b = true;
720 }
721 // unexpected situation since counter can not be KVED_FLASH_UINT_MAX: we
722 // will invalidate the sector. Maybe some people would like to analize it
723 // and rescue some valid entries. Not implemented, anyway.
724 else if((cnt_sec_a == KVED_FLASH_UINT_MAX) || (cnt_sec_b == KVED_FLASH_UINT_MAX))
725 {
726 invalidate_a = cnt_sec_a == KVED_FLASH_UINT_MAX ;
727 invalidate_b = cnt_sec_b == KVED_FLASH_UINT_MAX;
728 }
729
730 // ok, regular situation where one counter is the newest and
731 // all values are normal, just choose the newest one
732 if((invalidate_a == false) && (invalidate_b == false))
733 {
734 if(cnt_sec_a > cnt_sec_b)
735 invalidate_b = true;
736 else
737 invalidate_a = true;
738 }
739
740 // Invalidate selected sectors ...
741 if(invalidate_a)
742 kved_flash_data_write(KVED_FLASH_SECTOR_A,0,0);
743
744 if(invalidate_b)
745 kved_flash_data_write(KVED_FLASH_SECTOR_B,0,0);
746 }
747}

749static void kved_data_consistency_check(void)
750{
751 for(uint16_t index = ctrl.first_index ; index <= ctrl.last_index ; index += KVED_ENTRY_SIZE_IN_WORDS)
752 {
753 kved_word_t key = kved_flash_data_read(ctrl.sector,index);
754 kved_word_t val = kved_flash_data_read(ctrl.sector,index + 1);
755
756 // As we write value first and key after and we may be powered off during this operation
757 // it is necessary to fix cases where only the value was written. In such situation
758 // a key with only FFs may exist and this entry will be deleted.
759 // If the power down occurs when key is been written it is not possible to
760 // detect if the operation was finished or not. But this case need to be solved
761 // by application, removing any unknown or unexpected key (application knows its owns keys, kved not).
762 if((key == KVED_FLASH_UINT_MAX) && (val != KVED_FLASH_UINT_MAX))
763 {
764 kved_flash_data_write(ctrl.sector,index,0);
765 ctrl.stats.num_deleted_entries++;
766 ctrl.stats.num_free_entries--;
767 continue;
768 }
769
770 // Second possible situation: new data is written in the same section and the
771 // old value is not erased due some power down.
772 // So, it is required to check for duplicated keys AHEAD (always)
773 // and erase the old entry.
774 if(kved_is_valid_key(key))
775 {
776 for(uint16_t dup_key_index = index + KVED_ENTRY_SIZE_IN_WORDS ; dup_key_index <= ctrl.last_index ; dup_key_index += KVED_ENTRY_SIZE_IN_WORDS)
777 {
778 kved_word_t dup_key = kved_flash_data_read(ctrl.sector,dup_key_index);
779 if(kved_is_valid_key(dup_key))
780 {
781 if(KVED_HDR_MASK_KEY(dup_key) == KVED_HDR_MASK_KEY(key))
782 {
783 kved_flash_data_write(ctrl.sector,index,0);
784 ctrl.stats.num_deleted_entries++;
785 ctrl.stats.num_used_entries--;
786 break;
787 }
788 }
789 }
790 }
791 }

801void kved_init(void)
802{
803 kved_flash_init();
804
805 kved_sector_consistency_check();
806
807 kved_word_t id_sec_a = kved_flash_data_read(KVED_FLASH_SECTOR_A,0);
808 kved_word_t id_sec_b = kved_flash_data_read(KVED_FLASH_SECTOR_B,0);
809
810 if(id_sec_a == KVED_SIGNATURE_ENTRY)
811 {
812 ctrl.sector = KVED_FLASH_SECTOR_A;
813 }
814 else if(id_sec_b == KVED_SIGNATURE_ENTRY)
815 {
816 ctrl.sector = KVED_FLASH_SECTOR_B;
817 }
818 else
819 {
820 ctrl.sector = KVED_FLASH_SECTOR_A;
821 kved_flash_sector_erase(ctrl.sector);
822 kved_flash_data_write(ctrl.sector,1,0);// first cnt, after ID
823 kved_flash_data_write(ctrl.sector,0,KVED_SIGNATURE_ENTRY);
824 }
825
826 kved_sector_stats_read(&ctrl);
827
828 kved_data_consistency_check();
829
830 started = true;
831
832#ifdef KVED_DEBUG
833 kved_dump();
834#endif
835}
